```javascript
function exportElementHTML(element) {
  const html = element.outerHTML;
  const blob = new Blob([html], { type: 'text/html' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'element-export.html';
  a.click();
  URL.revokeObjectURL(url);
}

function exportElementCSS(element) {
  const computedStyle = window.getComputedStyle(element);
  let css = element.tagName.toLowerCase() + ' {' + '\n';
  for (let i = 0; i < computedStyle.length; i++) {
    const prop = computedStyle[i];
    const val = computedStyle.getPropertyValue(prop);
    if (!prop.startsWith('-') && !['content', 'webkitAppearance'].includes(prop)) {
      css += '  ' + prop + ': ' + val + ';\n';
    }
  }
  css += '}';
  const blob = new Blob([css], { type: 'text/css' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'element-export.css';
  a.click();
  URL.revokeObjectURL(url);
}

function copyElementHTML(element) {
  navigator.clipboard.writeText(element.outerHTML).then(() => {
    showToast('Copied to clipboard!', 'success');
  });
}
```

**Files:**

```javascript
// In IIFE state vars:
let currentTheme = localStorage.getItem('awe-theme') || 'dark';

// Apply theme function:
function applyTheme(theme) {
  currentTheme = theme;
  localStorage.setItem('awe-theme', theme);
  
  // Toggle CSS variables or class on panel
  const panel = document.getElementById('awe-editor-panel');
  const overlay = document.getElementById('awe-selection-overlay');
  if (theme === 'light') {
    panel.style.background = '#ffffff';
    panel.style.color = '#1a1a2e';
    overlay.style.background = 'rgba(0,0,0,0.1)';
    // Update all text colors inside panel
    document.querySelectorAll('#awe-editor-panel').forEach(el => {
      el.style.setProperty('--bg-primary', '#ffffff');
      el.style.setProperty('--text-primary', '#1a1a2e');
    });
  } else {
    panel.style.background = '#1a1a2e';
    panel.style.color = '#e2e8f0';
    overlay.style.background = 'rgba(255,255,255,0.1)';
  }
}

// Toggle function:
function toggleTheme() {
  const newTheme = currentTheme === 'dark' ? 'light' : 'dark';
  applyTheme(newTheme);
  // Also save to storage for sync across components
  chrome.storage.sync.set({ aweTheme: newTheme });
}
```

```javascript
async function getDailyUsage() {
  const result = await new Promise(resolve => chrome.storage.sync.get(['dailyUsage', 'lastResetDate'], resolve));
  const today = new Date().toDateString();
  if (result.lastResetDate !== today) {
    // Reset count for new day
    chrome.storage.sync.set({ dailyUsage: 0, lastResetDate: today });
    return 0;
  }
  return result.dailyUsage || 0;
}

async function incrementDailyUsage() {
  const current = await getDailyUsage();
  chrome.storage.sync.set({ dailyUsage: current + 1 });
  return current + 1;
}
```
